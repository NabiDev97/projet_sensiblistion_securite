<script>
  const simulateBtn = document.getElementById('simulateBtn');
  const warningBox = document.getElementById('warningBox');
  const advice = document.getElementById('advice');
  const showTips = document.getElementById('showTips');
  const quizBtn = document.getElementById('quizBtn');
  const quizArea = document.getElementById('quizArea');
  const submitQuiz = document.getElementById('submitQuiz');
  const quizResult = document.getElementById('quizResult');

  simulateBtn.addEventListener('click', () => {
    warningBox.style.display = 'block';
    advice.style.display = 'block';
    window.scrollTo({ top: warningBox.offsetTop - 80, behavior: 'smooth' });
  });

  showTips.addEventListener('click', () => {
    advice.style.display = advice.style.display === 'block' ? 'none' : 'block';
  });

  quizBtn.addEventListener('click', () => {
    quizArea.style.display = quizArea.style.display === 'block' ? 'none' : 'block';
    window.scrollTo({ top: quizArea.offsetTop - 80, behavior: 'smooth' });
  });

  // quiz logic
  submitQuiz.addEventListener('click', () => {
    const questions = document.querySelectorAll('.q');
    let score = 0;
    questions.forEach((q, idx) => {
      const correct = parseInt(q.getAttribute('data-answer'),10);
      const selectedBtn = q.querySelector('.choices button.selected');
      const choice = selectedBtn ? parseInt(selectedBtn.getAttribute('data-choice'),10) : null;
      if(choice === correct) score++;
      // mark buttons
      q.querySelectorAll('.choices button').forEach(btn => {
        btn.classList.remove('correct','wrong');
        const val = parseInt(btn.getAttribute('data-choice'),10);
        if(val === correct) btn.classList.add('correct');
        if(selectedBtn && btn === selectedBtn && val !== correct) btn.classList.add('wrong');
      });
    });
    quizResult.style.display = 'block';
    quizResult.innerHTML = `<strong>Résultat :</strong> ${score} / ${questions.length} bonnes réponses.<br>
      ${score === questions.length ? '<span style="color:var(--ok)">Très bien — vous avez les bons réflexes.</span>' : '<span style="color:var(--bad)">Pensez à appliquer la checklist avant d\'agir.</span>'}`;
    window.scrollTo({ top: quizResult.offsetTop - 80, behavior: 'smooth' });
  });

  // choice selection
  document.querySelectorAll('.choices').forEach(group => {
    group.addEventListener('click', e => {
      if(e.target.tagName !== 'BUTTON') return;
      group.querySelectorAll('button').forEach(b => b.classList.remove('selected'));
      e.target.classList.add('selected');
    });
  });
</script>
